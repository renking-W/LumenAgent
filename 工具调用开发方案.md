# LumenAgent 工具调用开发方案

## 1. 设计目标

| 目标 | 说明 |
|------|------|
| **模型无关** | 工具定义和 Agent 循环不依赖任何具体模型的格式。消息使用统一内部格式（content blocks），由模型适配器负责格式转换 |
| **工具即插件** | 每个工具是一个异步 Python 类，通过注册表管理，可热插拔 |
| **前端透明** | 工具执行以 SSE 事件通知前端，前端只需渲染工具卡片 |
| **防无限循环** | 内置调用频次保护，防止模型反复调用同一工具 |
| **安全沙箱** | 工具执行受工作目录、超时、路径校验等多层限制 |

## 3. 阶段一：工具定义系统

### 3.1 `ToolResult`

```python
# agent/tools/base.py

from dataclasses import dataclass
from typing import Any


@dataclass
class ToolResult:
    """工具执行结果（模型无关）。"""
    status: str        # "success" | "error"
    result: Any        # 执行结果（字符串或结构化数据）
    execution_time: float = 0
    is_error: bool = False

    @staticmethod
    def success(result: Any) -> "ToolResult":
        return ToolResult("success", result)

    @staticmethod
    def error(message: str) -> "ToolResult":
        return ToolResult("error", message, is_error=True)
```

### 3.2 `BaseTool`

```python
# agent/tools/base.py


class BaseTool:
    """
    所有工具的基类。类属性即 JSON Schema 声明。

    子类只需要定义 name / description / parameters 三个类属性
    并实现 execute() 方法。
    """

    name: str = ""
    description: str = ""
    # JSON Schema 格式（与模型无关的统一内部格式）
    parameters: dict = {
        "type": "object",
        "properties": {},
        "required": [],
    }

    async def execute(self, params: dict) -> ToolResult:
        """子类实现具体逻辑。异步便于 IO 密集型工具。"""
        raise NotImplementedError

    def to_internal_schema(self) -> dict:
        """转换为统一内部格式的工具定义（供 ModelAdapter 使用）。"""
        return {
            "name": self.name,
            "description": self.description,
            "input_schema": self.parameters,
        }
```

**使用示例（Read 工具）：**

### 3.3 `ToolRegistry`

```python
# agent/tools/registry.py


class ToolRegistry:
    """
    工具注册表（单例）。

    管理所有可用工具的类信息。Agent 运行时通过它实例化工具。
    支持条件注册（例如 web_search 需要 API key 才注册）。
    """

    _tools: dict[str, type[BaseTool]] = {}

    @classmethod
    def register(cls, tool_cls: type[BaseTool]) -> type[BaseTool]:
        """装饰器：注册工具类。"""
        cls._tools[tool_cls.name] = tool_cls
        return tool_cls

    @classmethod
    def create_tool(cls, name: str) -> BaseTool | None:
        """按名称实例化工具。"""
        tool_cls = cls._tools.get(name)
        return tool_cls() if tool_cls else None

    @classmethod
    def create_all_tools(cls) -> list[BaseTool]:
        """实例化所有已注册工具。"""
        return [tool_cls() for tool_cls in cls._tools.values()]

    @classmethod
    def list_internal_schemas(cls) -> list[dict]:
        """返回统一内部格式的工具定义列表。"""
        return [tool_cls().to_internal_schema() for tool_cls in cls._tools.values()]

    @classmethod
    def clear(cls):
        """清空注册表（主要用于测试）。"""
        cls._tools.clear()
```

**注册方式（装饰器）：**


## 4. 阶段二：Agent 工具循环（核心）

### 4.1 `AgentStreamExecutor`

```python
# agent/agent.py

class AgentStreamExecutor:
    """
    流式 Agent 工具循环（完全模型无关）。

    职责：
    - 接收 ModelAdapter（处理模型格式转换）和工具列表
    - 执行多轮 tool_use/tool_result 循环
    - yield 事件供 SSE 推送
    - 防无限循环保护
    """

    def __init__(
        self,
        adapter: ModelAdapter,          # 模型适配器（负责格式转换）
        tools: list[BaseTool],          # 可用工具实例列表
        settings: Settings,
        *,
        max_turns: int = 20,
        on_event: Callable | None = None,
    ):
        self.adapter = adapter
        self.tools = {t.name: t for t in tools}
        self.max_turns = max_turns
        self.on_event = on_event
        self.settings = settings
        self.guard = ToolExecutionGuard()
```

### 4.2 主循环逻辑

```python
async def run_stream(
    self,
    messages: list[dict],   # 已加载的消息历史（内部格式）
) -> AsyncIterator[tuple[str, str | dict]]:
    """
    主入口：启动工具循环，yield 事件流。

    yield (kind, data):
      ("content", text)                    → 文本增量
      ("reasoning_content", text)          → 思维链增量（DeepSeek thinking 模式）
      ("tool_calls", [{"name", "id"},...]) → 本轮模型发起的工具列表
      ("tool_execution_start", {...})      → 单个工具开始执行
      ("tool_execution_end", {...})        → 单个工具执行完毕
      ("done", final_text)                 → 最终回复
      ("error", error_message)             → 错误
    """
```

```
伪码流程：

messages ← 已加载的消息历史

for turn in range(max_turns):
    # 1. 验证消息完整性
    messages = ensure_message_integrity(messages)

    # 2. 调用模型（流式）
    tool_calls_in_this_turn = []
    full_content = ""
    async for block in adapter.chat_stream(messages, tools_schemas):
        if block.type == "text":
            full_content += block.text
            yield ("content", block.text)
        elif block.type == "thinking":
            yield ("reasoning_content", block.thinking)
        elif block.type == "tool_use":
            tool_calls_in_this_turn.append(block)
        elif block.type == "stop_reason":
            stop_reason = block.reason

    # 3. 将 assistant 响应追加到消息历史
    assistant_blocks = []
    if full_content:
        assistant_blocks.append({"type": "text", "text": full_content})
    for tc in tool_calls_in_this_turn:
        assistant_blocks.append({"type": "tool_use", **tc})
    messages.append({"role": "assistant", "content": assistant_blocks})

    # 4. 没有工具调用 → 结束
    if not tool_calls_in_this_turn:
        yield ("done", full_content)
        return

    # 5. 通知前端
    yield ("tool_calls", [{"name": tc.name, "id": tc.id} for tc in tool_calls_in_this_turn])

    # 6. 逐个执行工具
    tool_result_blocks = []
    for tc in tool_calls_in_this_turn:
        # 防无限循环检查
        guard_result = self.guard.check(tc.name, tc.input)
        if not guard_result.allowed:
            tool_result_blocks.append({
                "type": "tool_result",
                "tool_use_id": tc.id,
                "content": guard_result.reason,
                "is_error": True,
            })
            yield ("tool_execution_start", {"name": tc.name, "arguments": tc.input})
            yield ("tool_execution_end", {"name": tc.name, "status": "error", ...})
            if guard_result.is_critical:
                yield ("error", guard_result.reason)
                return
            continue

        yield ("tool_execution_start", {"name": tc.name, "arguments": tc.input})

        start = time.time()
        tool = self.tools.get(tc.name)
        if not tool:
            result = ToolResult.error(f"Tool '{tc.name}' not found")
        else:
            result = await tool.execute(tc.input)

        elapsed = time.time() - start
        self.guard.record(tc.name, tc.input, result.status == "success")

        yield ("tool_execution_end", {
            "name": tc.name,
            "status": result.status,
            "execution_time": elapsed,
            "result_preview": str(result.result)[:200] if result.result else "",
        })

        tool_result_blocks.append({
            "type": "tool_result",
            "tool_use_id": tc.id,
            "content": str(result.result) if result.result else "",
            "is_error": result.is_error,
        })

    # 7. 追加 tool_result 到消息历史
    messages.append({"role": "user", "content": tool_result_blocks})

# 超出 max_turns
yield ("error", f"已执行 {max_turns} 轮，达到最大步数限制，请重试")
```

```python
def _execute_single_tool(self, tool_call: dict) -> ToolResult:
```

```python
# application/chat_service.py

async def reply_with_agent(
    repo: ConversationRepositoryPort,
    adapter: ModelAdapter,
    session_id: str,
    user_message: str,
    settings: Settings,
) -> AsyncIterator[tuple[str, str | dict]]:
    """用 Agent 工具循环处理会话请求。"""
    # 1. 用户消息落库
    content_blocks = json.dumps([{"type": "text", "text": user_message}], ensure_ascii=False)
    await repo.append_message(session_id, "user", content_blocks)

    # 2. 加载历史消息
    summary, recent = await _load_context(repo, session_id)
    # 解析为内部格式
    messages = _build_messages_from_history(summary, recent, user_message)

    # 3. 创建 Agent 执行器
    agent = AgentStreamExecutor(
        adapter=adapter,
        tools=ToolRegistry.create_all_tools(),
        settings=settings,
    )

    # 4. 运行工具循环
    final_text = ""
    async for kind, data in agent.run_stream(messages):
        if kind == "content":
            final_text += data
        elif kind == "done":
            final_text = data
        yield (kind, data)

    # 5. 助手回复落库
    if final_text.strip():
        assistant_blocks = json.dumps([{"type": "text", "text": final_text}], ensure_ascii=False)
        await repo.append_message(session_id, "assistant", assistant_blocks)
        new_count = await repo.increment_round_counter(session_id)
        asyncio.create_task(maybe_trigger_summary(repo, adapter, session_id, settings))
```

### 5.2 Read 工具

```python
# agent/tools/read.py

WORKSPACE_DIR: Path  # 由 Settings 配置
MAX_FILE_SIZE = 1 * 1024 * 1024  # 1MB

@ToolRegistry.register
class Read(BaseTool):
    name = "read"
    description = "Read or inspect file contents from the workspace directory"
    parameters = {
        "type": "object",
        "properties": {
            "path": {"type": "string", "description": "File path, relative to workspace"},
            "offset": {"type": "integer", "description": "Starting line (0-indexed)"},
            "limit": {"type": "integer", "description": "Max lines to return"},
        },
        "required": ["path"],
    }

    async def execute(self, params: dict) -> ToolResult:
        raw_path = params["path"]
        # 安全检查：防目录穿越
        full_path = self._resolve_safe_path(raw_path)
        if full_path is None:
            return ToolResult.error(f"Path '{raw_path}' is outside workspace")

        if not full_path.exists():
            return ToolResult.error(f"File '{raw_path}' not found")

        if full_path.is_dir():
            return ToolResult.success("\n".join(os.listdir(full_path)))

        if full_path.stat().st_size > MAX_FILE_SIZE:
            return ToolResult.error(f"File too large ({full_path.stat().st_size} bytes > 1MB)")

        # 读取文件（支持 offset/limit）
        with open(full_path, "r", encoding="utf-8", errors="replace") as f:
            lines = f.readlines()
        offset = params.get("offset", 0)
        limit = params.get("limit", len(lines))
        content = "".join(lines[offset:offset + limit])
        return ToolResult.success(content)

    def _resolve_safe_path(self, raw_path: str) -> Path | None:
        """安全解析路径，防目录穿越。"""
        try:
            target = (WORKSPACE_DIR / raw_path).resolve()
            target.relative_to(WORKSPACE_DIR.resolve())
            return target
        except (ValueError, OSError):
            return None
```

```python
# agent/tools/write.py

SENSITIVE_FILES = {".env", ".git", "config.py", "*.pem", "*.key"}

@ToolRegistry.register
class Write(BaseTool):
    name = "write"
    description = "Write content to a file in the workspace directory"
    parameters = {
        "type": "object",
        "properties": {
            "path": {"type": "string", "description": "File path, relative to workspace"},
            "content": {"type": "string", "description": "Content to write"},

        "required": ["path", "content"],

        if full_path.name in SENSITIVE_FILES:

        full_path.parent.mkdir(parents=True, exist_ok=True)
        with open(full_path, "w", encoding="utf-8") as f:
            f.write(params["content"])
        return ToolResult.success(f"Written {len(params['content'])} bytes to {raw_path}")

    def _resolve_safe_path(self, raw_path: str) -> Path | None:
        try:
            target = (WORKSPACE_DIR / raw_path).resolve()
            target.relative_to(WORKSPACE_DIR.resolve())
            return target
        except (ValueError, OSError):
            return None
```

### 5.4 Bash 工具

```python
# agent/tools/bash.py

@ToolRegistry.register
class Bash(BaseTool):
    name = "bash"
    description = "Execute a shell command in the workspace directory"
    parameters = {
        "type": "object",
        "properties": {
            "command": {"type": "string", "description": "Shell command to execute"},
            "timeout": {"type": "integer", "description": "Timeout in seconds (default 30)"},
        },
        "required": ["command"],
    }

    async def execute(self, params: dict) -> ToolResult:
        command = params["command"]
        timeout = min(params.get("timeout", 30), 60)
        try:
            proc = await asyncio.create_subprocess_shell(
                command,
                stdout=asyncio.subprocess.PIPE,
                stderr=asyncio.subprocess.PIPE,
                cwd=WORKSPACE_DIR,
            )
            stdout, stderr = await asyncio.wait_for(
                proc.communicate(), timeout=timeout
            )
            output = stdout.decode("utf-8", errors="replace")
            if stderr:
                output += "\n[STDERR]\n" + stderr.decode("utf-8", errors="replace")
            if proc.returncode != 0 and not output.strip():
                return ToolResult.error(f"Exit code {proc.returncode}")
            return ToolResult.success(output.strip() or f"(exit code {proc.returncode})")
        except asyncio.TimeoutError:
            return ToolResult.error(f"Command timed out after {timeout}s")
        except Exception as e:
            return ToolResult.error(str(e))
```

### 5.5 WebFetch 工具

```python
# agent/tools/web_fetch.py

_BLOCKED_IPS = ("127.", "10.", "192.168.", "172.16.", "0.", "169.254.")
_MAX_RESPONSE_SIZE = 1 * 1024 * 1024  # 1MB
_REQUEST_TIMEOUT = 15

@ToolRegistry.register
class WebFetch(BaseTool):
    name = "web_fetch"
    description = "Fetch and read content from a URL (web pages, APIs, etc.)"
    parameters = {
        "type": "object",
        "properties": {
            "url": {"type": "string", "description": "URL to fetch"},
            "max_size": {"type": "integer", "description": "Max response size in bytes"},
        },
        "required": ["url"],
    }

    async def execute(self, params: dict) -> ToolResult:
        url = params["url"]
        max_size = min(params.get("max_size", _MAX_RESPONSE_SIZE), _MAX_RESPONSE_SIZE)

        # 安全：禁止内网
        from urllib.parse import urlparse
        host = urlparse(url).hostname or ""
        if any(host.startswith(prefix) for prefix in _BLOCKED_IPS):
            return ToolResult.error(f"Blocked internal IP: {host}")

        try:
            async with httpx.AsyncClient(timeout=_REQUEST_TIMEOUT) as client:
                response = await client.get(url, follow_redirects=True)
                response.raise_for_status()
                content = response.text[:max_size]
                return ToolResult.success(content)
        except httpx.HTTPStatusError as e:
            return ToolResult.error(f"HTTP {e.response.status_code}")
        except httpx.RequestError as e:
            return ToolResult.error(f"Request failed: {e}")
```

```python
    def trim_to_max_turns(self, messages: list[dict]) -> list[dict]:
        """保留最近 N 轮，丢弃更早轮次。"""
        turns = self.identify_complete_turns(messages)
        if len(turns) <= self.max_turns:
            return messages
        keep = turns[-self.max_turns:]
        return [msg for turn in keep for msg in turn]
```

### 7.2 Tool Result 截断

```python
def truncate_tool_results(self, messages: list[dict]) -> list[dict]:
    """截断超过上限的 tool_result 块内容。"""
    for msg in messages:
        for block in msg.get("content", []):
            if block.get("type") == "tool_result" and isinstance(block.get("content"), str):
                if len(block["content"]) > self.max_tool_result_chars:
                    block["content"] = block["content"][:self.max_tool_result_chars] \
                        + f"\n...[truncated {len(block['content'])} chars]"
    return messages
```

### 7.3 溢出恢复

```python
async def recover_from_overflow(
    messages: list[dict], adapter: ModelAdapter
) -> list[dict]:
    """
    API 返回 context overflow 错误时的逐级恢复。

    第 1 级：截断 tool_result 到 10K
    第 2 级：丢弃最早 50% 轮次
    第 3 级：放弃
    """
    # 第 1 级
    messages = ContextManager().truncate_tool_results(messages)
    if await _try_request(messages, adapter):
        return messages

    # 第 2 级
    half = len(messages) // 2
    messages = messages[half:]
    if await _try_request(messages, adapter):
        return messages

    # 第 3 级
    raise RuntimeError("对话上下文已满，请重新开始")
```

### 7.4 防无限循环保护

```python
# agent/context.py

@dataclass
class ToolGuardResult:
    allowed: bool
    reason: str = ""
    is_critical: bool = False


class ToolExecutionGuard:
    """
    防无限循环保护。

    防止模型反复调用同一工具或在同一个参数上卡死。
    """

    MAX_SAME_ARGS = 5      # 同工具+同参数 ≥5 次 → 拦截
    MAX_TOOL_FAILURES = 6  # 同工具连续失败 ≥6 次 → 拦截
    MAX_TOTAL_FAILURES = 8 # 总计连续失败 ≥8 次 → 硬中断

    def __init__(self):
        self._history: list[tuple[str, str, bool]] = []  # (name, args_hash, success)

    def check(self, tool_name: str, args: dict) -> ToolGuardResult:
        args_hash = self._hash_args(args)

        # 1. 同参数重复调用
        same_args = sum(1 for n, h, _ in reversed(self._history)
                        if n == tool_name and h == args_hash)
        if same_args >= self.MAX_SAME_ARGS:
            return ToolGuardResult(
                False,
                f"工具 '{tool_name}' 使用相同参数已被连续调用 {same_args} 次，已阻止",
            )

        # 2. 同工具连续失败
        same_tool_fails = sum(1 for n, _, s in reversed(self._history)
                              if n == tool_name and not s)
        if same_tool_fails >= self.MAX_TOOL_FAILURES:
            return ToolGuardResult(
                False,
                f"工具 '{tool_name}' 连续失败 {same_tool_fails} 次，已阻止",
            )

        # 3. 全局连续失败（硬中断）
        total_fails = sum(1 for _, _, s in reversed(self._history) if not s)
        if total_fails >= self.MAX_TOTAL_FAILURES:
            return ToolGuardResult(
                False,
                "连续失败次数过多，请换个方式描述需求或稍后再试",
                is_critical=True,
            )

        return ToolGuardResult(True)

    def record(self, tool_name: str, args: dict, success: bool):
        """记录工具执行结果。"""
        self._history.append((tool_name, self._hash_args(args), success))
        # 只保留最近 50 条
        if len(self._history) > 50:
            self._history = self._history[-50:]

    def _hash_args(self, args: dict) -> str:
        import hashlib, json
        return hashlib.md5(
            json.dumps(args, sort_keys=True, ensure_ascii=False).encode()
        ).hexdigest()[:8]
```
